% ============================================
% PAGE LAYOUT
% ============================================

% Headers and footers
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{} % Clear all headers/footers

% Header style
\fancyhead[LE]{\small\textit{\leftmark}} % Chapter on even pages
\fancyhead[RO]{\small\textit{Elemental Alchemy}} % Book title on odd pages

% Footer style (page numbers)
\fancyfoot[LE,RO]{\small\thepage}

% Remove header line
\renewcommand{\headrulewidth}{0pt}

% Chapter pages (no header)
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[LE,RO]{\small\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

% ============================================
% CHAPTER STYLING
% ============================================

\usepackage{titlesec}

% Chapter format
\titleformat{\chapter}[display]
  {\normalfont\huge\bfseries\sffamily}
  {\chaptertitlename\ \thechapter}
  {20pt}
  {\Huge\color{SandyGold}}

% Section format
\titleformat{\section}
  {\normalfont\Large\bfseries\sffamily\color{DarkEmerald}}
  {\thesection}
  {1em}
  {}

% Subsection format
\titleformat{\subsection}
  {\normalfont\large\bfseries\sffamily\color{SandyGold}}
  {\thesubsection}
  {1em}
  {}
